4// Implemented features:
5// [!] Renderer: User texture binding. Use 'VkDescriptorSet' as ImTextureID. Read the FAQ about ImTextureID! See https://github.com/ocornut/imgui/pull/914 for discussions.
6// [X] Renderer: Large meshes support (64k+ vertices) with 16-bit indices.
7
8// Important: on 32-bit systems, user texture binding is only supported if your imconfig file has '#define ImTextureID ImU64'.
9// This is because we need ImTextureID to carry a 64-bit value and by default ImTextureID is defined as void*.
10// To build this on 32-bit systems and support texture changes:
11// - [Solution 1] IDE/msbuild: in "Properties/C++/Preprocessor Definitions" add 'ImTextureID=ImU64' (this is what we do in our .vcxproj files)
12// - [Solution 2] IDE/msbuild: in "Properties/C++/Preprocessor Definitions" add 'IMGUI_USER_CONFIG="my_imgui_config.h"' and inside 'my_imgui_config.h' add '#define ImTextureID ImU64' and as many other options as you like.
13// - [Solution 3] IDE/msbuild: edit imconfig.h and add '#define ImTextureID ImU64' (prefer solution 2 to create your own config file!)
14// - [Solution 4] command-line: add '/D ImTextureID=ImU64' to your cl.exe command-line (this is what we do in our batch files)

429static void ImGui_ImplVulkan_SetupRenderState(ImDrawData* draw_data, VkPipeline pipeline, VkCommandBuffer command_buffer, ImGui_ImplVulkan_FrameRenderBuffers* rb, int fb_width, int fb_height)
430{
431 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
432
433 // Bind pipeline:
434 {
435 vkCmdBindPipeline(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, pipeline);
436 }
437
438 // Bind Vertex And Index Buffer:
439 if (draw_data->TotalVtxCount > 0)
440 {
441 VkBuffer vertex_buffers[1] = { rb->VertexBuffer };
442 VkDeviceSize vertex_offset[1] = { 0 };
443 vkCmdBindVertexBuffers(command_buffer, 0, 1, vertex_buffers, vertex_offset);
444 vkCmdBindIndexBuffer(command_buffer, rb->IndexBuffer, 0, sizeof(ImDrawIdx) == 2 ? VK_INDEX_TYPE_UINT16 : VK_INDEX_TYPE_UINT32);
445 }
446
447 // Setup viewport:
448 {
449 VkViewport viewport;
450 viewport.x = 0;
451 viewport.y = 0;
452 viewport.width = (float)fb_width;
453 viewport.height = (float)fb_height;
454 viewport.minDepth = 0.0f;
455 viewport.maxDepth = 1.0f;
456 vkCmdSetViewport(command_buffer, 0, 1, &viewport);
457 }
458
459 // Setup scale and translation:
460 // Our visible imgui space lies from draw_data->DisplayPps (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
461 {
462 float scale[2];
463 scale[0] = 2.0f / draw_data->DisplaySize.x;
464 scale[1] = 2.0f / draw_data->DisplaySize.y;
465 float translate[2];
466 translate[0] = -1.0f - draw_data->DisplayPos.x * scale[0];
467 translate[1] = -1.0f - draw_data->DisplayPos.y * scale[1];
468 vkCmdPushConstants(command_buffer, bd->PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 0, sizeof(float) * 2, scale);
469 vkCmdPushConstants(command_buffer, bd->PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 2, sizeof(float) * 2, translate);
470 }
471}

473// Render function
474void ImGui_ImplVulkan_RenderDrawData(ImDrawData* draw_data, VkCommandBuffer command_buffer, VkPipeline pipeline)
475{
476 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
477 int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
478 int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
479 if (fb_width <= 0 || fb_height <= 0)
480 return;
481
482 ImGui_ImplVulkan_Data* bd = ImGui_ImplVulkan_GetBackendData();
483 ImGui_ImplVulkan_InitInfo* v = &bd->VulkanInitInfo;
484 if (pipeline == VK_NULL_HANDLE)
485 pipeline = bd->Pipeline;
486
487 // Allocate array to store enough vertex/index buffers
488 ImGui_ImplVulkan_WindowRenderBuffers* wrb = &bd->MainWindowRenderBuffers;
489 if (wrb->FrameRenderBuffers == nullptr)
490 {
491 wrb->Index = 0;
492 wrb->Count = v->ImageCount;
493 wrb->FrameRenderBuffers = (ImGui_ImplVulkan_FrameRenderBuffers*)IM_ALLOC(sizeof(ImGui_ImplVulkan_FrameRenderBuffers) * wrb->Count);
494 memset(wrb->FrameRenderBuffers, 0, sizeof(ImGui_ImplVulkan_FrameRenderBuffers) * wrb->Count);
495 }
496 IM_ASSERT(wrb->Count == v->ImageCount);
497 wrb->Index = (wrb->Index + 1) % wrb->Count;
498 ImGui_ImplVulkan_FrameRenderBuffers* rb = &wrb->FrameRenderBuffers[wrb->Index];
499
500 if (draw_data->TotalVtxCount > 0)
501 {
502 // Create or resize the vertex/index buffers
503 size_t vertex_size = AlignBufferSize(draw_data->TotalVtxCount * sizeof(ImDrawVert), bd->BufferMemoryAlignment);
504 size_t index_size = AlignBufferSize(draw_data->TotalIdxCount * sizeof(ImDrawIdx), bd->BufferMemoryAlignment);
505 if (rb->VertexBuffer == VK_NULL_HANDLE || rb->VertexBufferSize < vertex_size)
506 CreateOrResizeBuffer(rb->VertexBuffer, rb->VertexBufferMemory, rb->VertexBufferSize, vertex_size, VK_BUFFER_USAGE_VERTEX_BUFFER_BIT);
507 if (rb->IndexBuffer == VK_NULL_HANDLE || rb->IndexBufferSize < index_size)
508 CreateOrResizeBuffer(rb->IndexBuffer, rb->IndexBufferMemory, rb->IndexBufferSize, index_size, VK_BUFFER_USAGE_INDEX_BUFFER_BIT);
509
510 // Upload vertex/index data into a single contiguous GPU buffer
511 ImDrawVert* vtx_dst = nullptr;
512 ImDrawIdx* idx_dst = nullptr;
513 VkResult err = vkMapMemory(v->Device, rb->VertexBufferMemory, 0, vertex_size, 0, (void**)&vtx_dst);
514 check_vk_result(err);
515 err = vkMapMemory(v->Device, rb->IndexBufferMemory, 0, index_size, 0, (void**)&idx_dst);
516 check_vk_result(err);
517 for (int n = 0; n < draw_data->CmdListsCount; n++)
518 {
519 const ImDrawList* cmd_list = draw_data->CmdLists[n];
520 memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
521 memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
522 vtx_dst += cmd_list->VtxBuffer.Size;
523 idx_dst += cmd_list->IdxBuffer.Size;
524 }
525 VkMappedMemoryRange range[2] = {};
526 range[0].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
527 range[0].memory = rb->VertexBufferMemory;
528 range[0].size = VK_WHOLE_SIZE;
529 range[1].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
530 range[1].memory = rb->IndexBufferMemory;
531 range[1].size = VK_WHOLE_SIZE;
532 err = vkFlushMappedMemoryRanges(v->Device, 2, range);
533 check_vk_result(err);
534 vkUnmapMemory(v->Device, rb->VertexBufferMemory);
535 vkUnmapMemory(v->Device, rb->IndexBufferMemory);
536 }
537
538 // Setup desired Vulkan state
539 ImGui_ImplVulkan_SetupRenderState(draw_data, pipeline, command_buffer, rb, fb_width, fb_height);
540
541 // Will project scissor/clipping rectangles into framebuffer space
542 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
543 ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
544
545 // Render command lists
546 // (Because we merged all buffers into a single one, we maintain our own offset into them)
547 int global_vtx_offset = 0;
548 int global_idx_offset = 0;
549 for (int n = 0; n < draw_data->CmdListsCount; n++)
550 {
551 const ImDrawList* cmd_list = draw_data->CmdLists[n];
552 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
553 {
554 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
555 if (pcmd->UserCallback != nullptr)
556 {
557 // User callback, registered via ImDrawList::AddCallback()
558 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
559 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
560 ImGui_ImplVulkan_SetupRenderState(draw_data, pipeline, command_buffer, rb, fb_width, fb_height);
561 else
562 pcmd->UserCallback(cmd_list, pcmd);
563 }
564 else
565 {
566 // Project scissor/clipping rectangles into framebuffer space
567 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
568 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
569
570 // Clamp to viewport as vkCmdSetScissor() won't accept values that are off bounds
571 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
572 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
573 if (clip_max.x > fb_width) { clip_max.x = (float)fb_width; }
574 if (clip_max.y > fb_height) { clip_max.y = (float)fb_height; }
575 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
576 continue;
577
578 // Apply scissor/clipping rectangle
579 VkRect2D scissor;
580 scissor.offset.x = (int32_t)(clip_min.x);
581 scissor.offset.y = (int32_t)(clip_min.y);
582 scissor.extent.width = (uint32_t)(clip_max.x - clip_min.x);
583 scissor.extent.height = (uint32_t)(clip_max.y - clip_min.y);
584 vkCmdSetScissor(command_buffer, 0, 1, &scissor);
585
586 // Bind DescriptorSet with font or user texture
587 VkDescriptorSet desc_set[1] = { (VkDescriptorSet)pcmd->TextureId };
588 if (sizeof(ImTextureID) < sizeof(ImU64))
589 {
590 // We don't support texture switches if ImTextureID hasn't been redefined to be 64-bit. Do a flaky check that other textures haven't been used.
591 IM_ASSERT(pcmd->TextureId == (ImTextureID)bd->FontDescriptorSet);
592 desc_set[0] = bd->FontDescriptorSet;
593 }
594 vkCmdBindDescriptorSets(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, bd->PipelineLayout, 0, 1, desc_set, 0, nullptr);
595
596 // Draw
597 vkCmdDrawIndexed(command_buffer, pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
598 }
599 }
600 global_idx_offset += cmd_list->IdxBuffer.Size;
601 global_vtx_offset += cmd_list->VtxBuffer.Size;
602 }
603
604 // Note: at this point both vkCmdSetViewport() and vkCmdSetScissor() have been called.
605 // Our last values will leak into user/application rendering IF:
606 // - Your app uses a pipeline with VK_DYNAMIC_STATE_VIEWPORT or VK_DYNAMIC_STATE_SCISSOR dynamic state
607 // - And you forgot to call vkCmdSetViewport() and vkCmdSetScissor() yourself to explicitly set that state.
608 // If you use VK_DYNAMIC_STATE_VIEWPORT or VK_DYNAMIC_STATE_SCISSOR you are responsible for setting the values before rendering.
609 // In theory we should aim to backup/restore those values but I am not sure this is possible.
610 // We perform a call to vkCmdSetScissor() to set back a full viewport which is likely to fix things for 99% users but technically this is not perfect. (See github #4644)
611 VkRect2D scissor = { { 0, 0 }, { (uint32_t)fb_width, (uint32_t)fb_height } };
612 vkCmdSetScissor(command_buffer, 0, 1, &scissor);
613}

759 copy_barrier[0].image = bd->FontImage;
760 copy_barrier[0].subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
761 copy_barrier[0].subresourceRange.levelCount = 1;
762 copy_barrier[0].subresourceRange.layerCount = 1;
763 vkCmdPipelineBarrier(bd->FontCommandBuffer, VK_PIPELINE_STAGE_HOST_BIT, VK_PIPELINE_STAGE_TRANSFER_BIT, 0, 0, nullptr, 0, nullptr, 1, copy_barrier);
764
765 VkBufferImageCopy region = {};
766 region.imageSubresource.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
767 region.imageSubresource.layerCount = 1;
768 region.imageExtent.width = width;
769 region.imageExtent.height = height;
770 region.imageExtent.depth = 1;
771 vkCmdCopyBufferToImage(bd->FontCommandBuffer, upload_buffer, bd->FontImage, VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL, 1, &region);
772
773 VkImageMemoryBarrier use_barrier[1] = {};
774 use_barrier[0].sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
775 use_barrier[0].srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
776 use_barrier[0].dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
777 use_barrier[0].oldLayout = VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL;
778 use_barrier[0].newLayout = VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL;
779 use_barrier[0].srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
780 use_barrier[0].dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
781 use_barrier[0].image = bd->FontImage;
782 use_barrier[0].subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
783 use_barrier[0].subresourceRange.levelCount = 1;
784 use_barrier[0].subresourceRange.layerCount = 1;
785 vkCmdPipelineBarrier(bd->FontCommandBuffer, VK_PIPELINE_STAGE_TRANSFER_BIT, VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT, 0, 0, nullptr, 0, nullptr, 1, use_barrier);
786 }
787
788 // Store our identifier
789 io.Fonts->SetTexID((ImTextureID)bd->FontDescriptorSet);
790
791 // End command buffer
792 VkSubmitInfo end_info = {};
793 end_info.sType = VK_STRUCTURE_TYPE_SUBMIT_INFO;
794 end_info.commandBufferCount = 1;
795 end_info.pCommandBuffers = &bd->FontCommandBuffer;
796 err = vkEndCommandBuffer(bd->FontCommandBuffer);
797 check_vk_result(err);
798 err = vkQueueSubmit(v->Queue, 1, &end_info, VK_NULL_HANDLE);
799 check_vk_result(err);
800
801 err = vkQueueWaitIdle(v->Queue);
802 check_vk_result(err);
803
804 vkDestroyBuffer(v->Device, upload_buffer, v->Allocator);
805 vkFreeMemory(v->Device, upload_buffer_memory, v->Allocator);
806
807 return true;
808}

1046bool ImGui_ImplVulkan_LoadFunctions(PFN_vkVoidFunction(*loader_func)(const char* function_name, void* user_data), void* user_data)
1047{
1048 // Load function pointers
1049 // You can use the default Vulkan loader using:
1050 // ImGui_ImplVulkan_LoadFunctions([](const char* function_name, void*) { return vkGetInstanceProcAddr(your_vk_isntance, function_name); });
1051 // But this would be equivalent to not setting VK_NO_PROTOTYPES.
1052#ifdef VK_NO_PROTOTYPES
1053#define IMGUI_VULKAN_FUNC_LOAD(func) \
1054 func = reinterpret_cast<decltype(func)>(loader_func(#func, user_data)); \
1055 if (func == nullptr) \
1056 return false;
1057 IMGUI_VULKAN_FUNC_MAP(IMGUI_VULKAN_FUNC_LOAD)
1058#undef IMGUI_VULKAN_FUNC_LOAD
1059
1060#ifdef IMGUI_IMPL_VULKAN_HAS_DYNAMIC_RENDERING
1061 // Manually load those two (see #5446)
1062 ImGuiImplVulkanFuncs_vkCmdBeginRenderingKHR = reinterpret_cast<PFN_vkCmdBeginRenderingKHR>(loader_func("vkCmdBeginRenderingKHR", user_data));
1063 ImGuiImplVulkanFuncs_vkCmdEndRenderingKHR = reinterpret_cast<PFN_vkCmdEndRenderingKHR>(loader_func("vkCmdEndRenderingKHR", user_data));
1064#endif
1065#else
1066 IM_UNUSED(loader_func);
1067 IM_UNUSED(user_data);
1068#endif
1069
1070 g_FunctionsLoaded = true;
1071 return true;
1072}

1233VkSurfaceFormatKHR ImGui_ImplVulkanH_SelectSurfaceFormat(VkPhysicalDevice physical_device, VkSurfaceKHR surface, const VkFormat* request_formats, int request_formats_count, VkColorSpaceKHR request_color_space)
1234{
1235 IM_ASSERT(g_FunctionsLoaded && "Need to call ImGui_ImplVulkan_LoadFunctions() if IMGUI_IMPL_VULKAN_NO_PROTOTYPES or VK_NO_PROTOTYPES are set!");
1236 IM_ASSERT(request_formats != nullptr);
1237 IM_ASSERT(request_formats_count > 0);
1238
1239 // Per Spec Format and View Format are expected to be the same unless VK_IMAGE_CREATE_MUTABLE_BIT was set at image creation
1240 // Assuming that the default behavior is without setting this bit, there is no need for separate Swapchain image and image view format
1241 // Additionally several new color spaces were introduced with Vulkan Spec v1.0.40,
1242 // hence we must make sure that a format with the mostly available color space, VK_COLOR_SPACE_SRGB_NONLINEAR_KHR, is found and used.
1243 uint32_t avail_count;
1244 vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, nullptr);
1245 ImVector<VkSurfaceFormatKHR> avail_format;
1246 avail_format.resize((int)avail_count);
1247 vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, avail_format.Data);
1248
1249 // First check if only one format, VK_FORMAT_UNDEFINED, is available, which would imply that any format is available
1250 if (avail_count == 1)
1251 {
1252 if (avail_format[0].format == VK_FORMAT_UNDEFINED)
1253 {
1254 VkSurfaceFormatKHR ret;
1255 ret.format = request_formats[0];
1256 ret.colorSpace = request_color_space;
1257 return ret;
1258 }
1259 else
1260 {
1261 // No point in searching another format
1262 return avail_format[0];
1263 }
1264 }
1265 else
1266 {
1267 // Request several formats, the first found will be used
1268 for (int request_i = 0; request_i < request_formats_count; request_i++)
1269 for (uint32_t avail_i = 0; avail_i < avail_count; avail_i++)
1270 if (avail_format[avail_i].format == request_formats[request_i] && avail_format[avail_i].colorSpace == request_color_space)
1271 return avail_format[avail_i];
1272
1273 // If none of the requested image formats could be found, use the first available
1274 return avail_format[0];
1275 }

150// Backend data stored in io.BackendPlatformUserData to allow support for multiple Dear ImGui contexts
151// It is STRONGLY preferred that you use docking branch with multi-viewports (== single Dear ImGui context + multiple windows) instead of multiple Dear ImGui contexts.
152// FIXME: multi-context support is not well tested and probably dysfunctional in this backend.
153// - Because glfwPollEvents() process all windows and some events may be called outside of it, you will need to register your own callbacks
154// (passing install_callbacks=false in ImGui_ImplGlfw_InitXXX functions), set the current dear imgui context and then call our callbacks.
155// - Otherwise we may need to store a GLFWWindow* -> ImGuiContext* map and handle this in the backend, adding a little bit of extra complexity to it.
156// FIXME: some shared resources (mouse cursor shape, gamepad) are mishandled when using multi-context.
157static ImGui_ImplGlfw_Data* ImGui_ImplGlfw_GetBackendData()
158{
159 return ImGui::GetCurrentContext() ? (ImGui_ImplGlfw_Data*)ImGui::GetIO().BackendPlatformUserData : nullptr;
160}

299// See https://github.com/ocornut/imgui/issues/6034 and https://github.com/glfw/glfw/issues/1630
300static void ImGui_ImplGlfw_UpdateKeyModifiers(GLFWwindow* window)
301{
302 ImGuiIO& io = ImGui::GetIO();
303 io.AddKeyEvent(ImGuiMod_Ctrl, (glfwGetKey(window, GLFW_KEY_LEFT_CONTROL) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_CONTROL) == GLFW_PRESS));
304 io.AddKeyEvent(ImGuiMod_Shift, (glfwGetKey(window, GLFW_KEY_LEFT_SHIFT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SHIFT) == GLFW_PRESS));
305 io.AddKeyEvent(ImGuiMod_Alt, (glfwGetKey(window, GLFW_KEY_LEFT_ALT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_ALT) == GLFW_PRESS));
306 io.AddKeyEvent(ImGuiMod_Super, (glfwGetKey(window, GLFW_KEY_LEFT_SUPER) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SUPER) == GLFW_PRESS));
307}

343static int ImGui_ImplGlfw_TranslateUntranslatedKey(int key, int scancode)
344{
345#if GLFW_HAS_GETKEYNAME && !defined(__EMSCRIPTEN__)
346 // GLFW 3.1+ attempts to "untranslate" keys, which goes the opposite of what every other framework does, making using lettered shortcuts difficult.
347 // (It had reasons to do so: namely GLFW is/was more likely to be used for WASD-type game controls rather than lettered shortcuts, but IHMO the 3.1 change could have been done differently)
348 // See https://github.com/glfw/glfw/issues/1502 for details.
349 // Adding a workaround to undo this (so our keys are translated->untranslated->translated, likely a lossy process).
350 // This won't cover edge cases but this is at least going to cover common cases.
351 if (key >= GLFW_KEY_KP_0 && key <= GLFW_KEY_KP_EQUAL)
352 return key;
353 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
354 const char* key_name = glfwGetKeyName(key, scancode);
355 glfwSetErrorCallback(prev_error_callback);
356#if GLFW_HAS_GETERROR && !defined(__EMSCRIPTEN__) // Eat errors (see #5908)
357 (void)glfwGetError(nullptr);
358#endif
359 if (key_name && key_name[0] != 0 && key_name[1] == 0)
360 {
361 const char char_names[] = "`-=[]\\,;\'./";
362 const int char_keys[] = { GLFW_KEY_GRAVE_ACCENT, GLFW_KEY_MINUS, GLFW_KEY_EQUAL, GLFW_KEY_LEFT_BRACKET, GLFW_KEY_RIGHT_BRACKET, GLFW_KEY_BACKSLASH, GLFW_KEY_COMMA, GLFW_KEY_SEMICOLON, GLFW_KEY_APOSTROPHE, GLFW_KEY_PERIOD, GLFW_KEY_SLASH, 0 };
363 IM_ASSERT(IM_ARRAYSIZE(char_names) == IM_ARRAYSIZE(char_keys));
364 if (key_name[0] >= '0' && key_name[0] <= '9') { key = GLFW_KEY_0 + (key_name[0] - '0'); }
365 else if (key_name[0] >= 'A' && key_name[0] <= 'Z') { key = GLFW_KEY_A + (key_name[0] - 'A'); }
366 else if (key_name[0] >= 'a' && key_name[0] <= 'z') { key = GLFW_KEY_A + (key_name[0] - 'a'); }
367 else if (const char* p = strchr(char_names, key_name[0])) { key = char_keys[p - char_names]; }
368 }
369 // if (action == GLFW_PRESS) printf("key %d scancode %d name '%s'\n", key, scancode, key_name);
370#else
371 IM_UNUSED(scancode);
372#endif
373 return key;

552static bool ImGui_ImplGlfw_Init(GLFWwindow* window, bool install_callbacks, GlfwClientApi client_api)
553{
554 ImGuiIO& io = ImGui::GetIO();
555 IM_ASSERT(io.BackendPlatformUserData == nullptr && "Already initialized a platform backend!");
556 //printf("GLFW_VERSION: %d.%d.%d (%d)", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION, GLFW_VERSION_COMBINED);
557
558 // Setup backend capabilities flags
559 ImGui_ImplGlfw_Data* bd = IM_NEW(ImGui_ImplGlfw_Data)();
560 io.BackendPlatformUserData = (void*)bd;
561 io.BackendPlatformName = "imgui_impl_glfw";
562 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
563 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
564
565 bd->Window = window;
566 bd->Time = 0.0;
567
568 io.SetClipboardTextFn = ImGui_ImplGlfw_SetClipboardText;
569 io.GetClipboardTextFn = ImGui_ImplGlfw_GetClipboardText;
570 io.ClipboardUserData = bd->Window;
571
572 // Create mouse cursors
573 // (By design, on X11 cursors are user configurable and some cursors may be missing. When a cursor doesn't exist,
574 // GLFW will emit an error which will often be printed by the app, so we temporarily disable error reporting.
575 // Missing cursors will return nullptr and our _UpdateMouseCursor() function will use the Arrow cursor instead.)
576 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
577 bd->MouseCursors[ImGuiMouseCursor_Arrow] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
578 bd->MouseCursors[ImGuiMouseCursor_TextInput] = glfwCreateStandardCursor(GLFW_IBEAM_CURSOR);
579 bd->MouseCursors[ImGuiMouseCursor_ResizeNS] = glfwCreateStandardCursor(GLFW_VRESIZE_CURSOR);
580 bd->MouseCursors[ImGuiMouseCursor_ResizeEW] = glfwCreateStandardCursor(GLFW_HRESIZE_CURSOR);
581 bd->MouseCursors[ImGuiMouseCursor_Hand] = glfwCreateStandardCursor(GLFW_HAND_CURSOR);
582#if GLFW_HAS_NEW_CURSORS
583 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_RESIZE_ALL_CURSOR);
584 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_RESIZE_NESW_CURSOR);
585 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_RESIZE_NWSE_CURSOR);
586 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_NOT_ALLOWED_CURSOR);
587#else
588 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
589 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
590 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
591 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
592#endif
593 glfwSetErrorCallback(prev_error_callback);
594#if GLFW_HAS_GETERROR && !defined(__EMSCRIPTEN__) // Eat errors (see #5908)
595 (void)glfwGetError(nullptr);
596#endif
597
598 // Chain GLFW callbacks: our callbacks will call the user's previously installed callbacks, if any.
599 if (install_callbacks)
600 ImGui_ImplGlfw_InstallCallbacks(window);
601 // Register Emscripten Wheel callback to workaround issue in Emscripten GLFW Emulation (#6096)
602 // We intentionally do not check 'if (install_callbacks)' here, as some users may set it to false and call GLFW callback themselves.
603 // FIXME: May break chaining in case user registered their own Emscripten callback?
604#ifdef __EMSCRIPTEN__
605 emscripten_set_wheel_callback(EMSCRIPTEN_EVENT_TARGET_DOCUMENT, nullptr, false, ImGui_ImplEmscripten_WheelCallback);
606#endif
607
608 // Set platform dependent data in viewport
609 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
610#ifdef _WIN32
611 main_viewport->PlatformHandleRaw = glfwGetWin32Window(bd->Window);
612#elif defined(__APPLE__)
613 main_viewport->PlatformHandleRaw = (void*)glfwGetCocoaWindow(bd->Window);
614#else
615 IM_UNUSED(main_viewport);
616#endif
617
618 // Windows: register a WndProc hook so we can intercept some messages.
619#ifdef _WIN32
620 bd->GlfwWndProc = (WNDPROC)::GetWindowLongPtrW((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC);
621 IM_ASSERT(bd->GlfwWndProc != nullptr);
622 ::SetWindowLongPtrW((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC, (LONG_PTR)ImGui_ImplGlfw_WndProc);
623#endif
624
625 bd->ClientApi = client_api;
626 return true;
627}

782void ImGui_ImplGlfw_NewFrame()
783{
784 ImGuiIO& io = ImGui::GetIO();
785 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
786 IM_ASSERT(bd != nullptr && "Did you call ImGui_ImplGlfw_InitForXXX()?");
787
788 // Setup display size (every frame to accommodate for window resizing)
789 int w, h;
790 int display_w, display_h;
791 glfwGetWindowSize(bd->Window, &w, &h);
792 glfwGetFramebufferSize(bd->Window, &display_w, &display_h);
793 io.DisplaySize = ImVec2((float)w, (float)h);
794 if (w > 0 && h > 0)
795 io.DisplayFramebufferScale = ImVec2((float)display_w / (float)w, (float)display_h / (float)h);
796
797 // Setup time step
798 // (Accept glfwGetTime() not returning a monotonically increasing value. Seems to happens on disconnecting peripherals and probably on VMs and Emscripten, see #6491, #6189, #6114, #3644)
799 double current_time = glfwGetTime();
800 if (current_time <= bd->Time)
801 current_time = bd->Time + 0.00001f;
802 io.DeltaTime = bd->Time > 0.0 ? (float)(current_time - bd->Time) : (float)(1.0f / 60.0f);
803 bd->Time = current_time;
804
805 ImGui_ImplGlfw_UpdateMouseData();
806 ImGui_ImplGlfw_UpdateMouseCursor();
807
808 // Update game controllers (if enabled and available)
809 ImGui_ImplGlfw_UpdateGamepads();
810}
